#definiujemy klasy suce do ekstracji cech z danych 

class Featurizer: #Definiuje interfejs dla obiektw, ktre generuj cechy z danych.
    
    def __init__(self, y_column, smiles_col='Drug', **kwargs): #y_column- nazwa kolumny z y, a smiles_col- nazwa kolumny ze SMILES
                                                               # **kwargs- przechowywanie innych argumentw przekazywanych do metody
        self.y_column = y_column           
        self.smiles_col = smiles_col       #Dziki self mona odwoa si do atrybutw i metod tej konkretnej instancji wewntrz klasy.
        self.__dict__.update(kwargs)
    
    
    def __call__(self, df):                #W tym przypadku metoda __call__ nie jest zdefiniowana, co oznacza, 
        raise NotImplementedError()        #e kada klasa dziedziczca po Featurizer musi j zaimplementowa, aby mona byo j wywoa. 


class CheatingFeaturizer(Featurizer):      #dziedziczy po klasie Featurizer
    
    
    def __call__(self, df):                # W metodzie __call__() tej klasy, dla kadego wiersza w danych wejciowych df, 
        
        predictions = []                   
        labels = []
        predictions_column_name = 'Y'
        
        for i, row in df.iterrows():
            predictions.append(row[predictions_column_name])      # Warto predykcji pobierana jest z kolumny 'Y',
            labels.append(row[self.y_column])                     # Warto etykiety z kolumny zdefiniowanej w klasie bazowej y_column.
            
        predictions = np.array(predictions)       #wartoci etykiet i predykcji s zwracane w postaci tablic numpy
        labels = np.array(labels)
        return predictions, labels
        
        

class ECFPFeaturizer(Featurizer):               #te dziedziczy po Featurizer
    
    def __init__(self, y_column, radius=2, length=1024, **kwargs):
        
        self.radius = radius                    #ustawiamy wartoci pl radius i length w obiekcie ECFPFeaturizer na wartoci podane przy tworzeniu obiektu.
        self.length = length
        super().__init__(y_column, **kwargs)    #wywoanie konstruktora klasy nadrzdnej Featurizer 
        
        #Dziki temu, do pola y_column w obiekcie ECFPFeaturizer zostaje przypisana warto argumentu y_column 
        #podanego przy tworzeniu obiektu, a wszystkie dodatkowe argumenty kwargs zostan zapisane w formie sownikowej pod atrybutami obiektu ECFPFeaturizer.
        
        
        
    def __call__(self, df):  #W tej metodzie obliczane s fingerprinty dla czsteczek na podstawie SMILES-w, 
                             #ktre znajduj si w kolumnie okrelonej przez argument smiles_col
        fingerprints = []
        labels = []
        for i, row in df.iterrows():
            y = row[self.y_column]
            smiles = row[self.smiles_col]
            mol = Chem.MolFromSmiles(smiles)
            fp = AllChem.GetMorganFingerprintAsBitVect(mol, self.radius, nBits=self.length) #Fingerprinty obliczane s z uyciem funkcji GetMorganFingerprintAsBitVect z moduu AllChem biblioteki RDKit, ktra tworzy reprezentacj molekularn czsteczki.
            fingerprints.append(fp)
            labels.append(y)
            
        fingerprints = np.array(fingerprints)
        labels = np.array(labels)
        return fingerprints, labels
